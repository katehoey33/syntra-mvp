# -----------------------------
# Page 1: Course Setup
# -----------------------------
if page == "1. Course Setup":
    render_header()

    st.markdown(
        """
        <div class="section-card">
            <h3>Start a New Semester</h3>
            <p>
            Set up your course context so the platform can compare student learning evidence against
            course goals, learning outcomes, assignments, rubrics, and professor feedback. This MVP does not grade,
            rank, or monitor students. It summarizes class-level learning patterns to support instructional decisions.
            </p>
        </div>
        """,
        unsafe_allow_html=True
    )

    left_col, right_col = st.columns([2, 1])

    with left_col:
        st.subheader("Course Criteria Setup")

        course_name = st.text_input(
            "Course Name",
            value=""
        )

        semester = st.selectbox(
            "Semester",
            ["Summer 2026", "Fall 2026", "Spring 2027"]
        )

        professor_name = st.text_input(
            "Professor Name",
            value="Professor Smith"
        )

        learning_outcomes = st.text_area(
            "Course Learning Outcomes",
            value=(
                "1. Understand the people-side barriers that shape change and innovation.\n"
                "2. Apply change management frameworks to organizational problems.\n"
                "3. Analyze innovation processes, structures, and strategies.\n"
                "4. Develop evidence-based recommendations for managing change and driving innovation."
            ),
            height=150
        )

        st.markdown("---")

        st.subheader("Connect Course Materials")

        brightspace_col, sync_col = st.columns([2, 1])

        with brightspace_col:
            st.markdown(
                """
                **Brightspace Connection**

                In the full version, this platform would connect directly to Brightspace to gather:
                syllabi, assignments, rubrics, grade outputs, professor feedback, and anonymous student survey responses.
                """
            )

        with sync_col:
            if st.button("Connect to Brightspace"):
                st.session_state.brightspace_connected = True
                st.session_state.last_sync = datetime.now().strftime("%B %d, %Y at %I:%M %p")

        if st.session_state.brightspace_connected:
            st.success(f"Brightspace connected. Last refreshed: {st.session_state.last_sync}")
        else:
            st.warning("Brightspace not connected yet.")

        st.markdown("### Course Context Uploads")
        st.caption("These uploads represent course materials that would be connected through Brightspace in a full implementation.")

        syllabus_file = st.file_uploader("Upload Syllabus", type=["pdf", "docx", "txt"])

        if syllabus_file is not None:
            st.success(f"Syllabus uploaded successfully: {syllabus_file.name}")
            st.info(
                "System update: Course topics, weekly modules, and learning objectives are now available for comparison."
            )

        materials_file = st.file_uploader("Upload Learning Materials", type=["pdf", "docx", "pptx", "txt", "zip"])

        if materials_file is not None:
            st.success(f"Learning materials uploaded successfully: {materials_file.name}")
            st.info(
                "System update: Readings, slides, and instructional materials are now available as course context."
            )

        assignments_file = st.file_uploader("Upload Assignments / Rubrics", type=["pdf", "docx", "csv", "xlsx", "zip"])

        if assignments_file is not None:
            st.success(f"Assignments and rubrics uploaded successfully: {assignments_file.name}")
            st.info(
                "System update: Assignment expectations and rubric criteria are now available for learning gap analysis."
            )

    with right_col:
        st.subheader("Course Setup Status")

        setup_items = {
            "Course Name": bool(course_name),
            "Semester Selected": bool(semester),
            "Professor Name": bool(professor_name),
            "Learning Outcomes Added": bool(learning_outcomes),
            "Brightspace Connected": st.session_state.brightspace_connected,
            "Syllabus Available": syllabus_file is not None,
            "Learning Materials Available": materials_file is not None,
            "Assignments / Rubrics Available": assignments_file is not None,
        }

        completed = sum(setup_items.values())
        total = len(setup_items)

        st.metric("Setup Progress", f"{completed}/{total}")

        progress_value = completed / total
        st.progress(progress_value)

        for item, status in setup_items.items():
            if status:
                st.markdown(f"✅ **{item}**")
            else:
                st.markdown(f"⚠️ **{item}**")

        st.markdown("---")

        st.markdown(
            """
            <div class="small-card">
                <h4>Responsible Use Reminder</h4>
                <p>
                This system supports faculty insight only. It does not assign grades, rank students,
                or generate disciplinary recommendations.
                </p>
            </div>
            """,
            unsafe_allow_html=True
        )

    st.markdown("---")
    st.subheader("Next Step")

    if completed >= 5:
        st.success(
            "Course setup is ready for Brightspace learning evidence refresh."
        )
    else:
        st.warning(
            "Complete the course setup criteria before running the Brightspace learning evidence refresh."
        )

    st.markdown(
        """
        <div class="footer-note">
        MVP Page 1: Professor course setup. Next page: Brightspace data refresh for assignment outcomes,
        professor feedback, and anonymous student reflections.
        </div>
        """,
        unsafe_allow_html=True
    )

    render_header()

        """

        unsafe_allow_html=True
    )

    left_col, right_col = st.columns([2, 1])

    with left_col:
        st.subheader("Refresh Brightspace Evidence")

        st.markdown(
            """
            This step simulates a secure Brightspace data connection. In the full implementation,
            the platform would retrieve existing course evidence from Brightspace, including assignment
            records, professor feedback, and anonymous student survey responses.
            """
        )

        if st.button("Refresh Brightspace Data"):
            st.session_state.brightspace_connected = True
            st.session_state.brightspace_data_loaded = True
            st.session_state.last_sync = datetime.now().strftime("%B %d, %Y at %I:%M %p")
            st.session_state.brightspace_df = load_mock_brightspace_data()

        if st.session_state.brightspace_data_loaded:
            df = st.session_state.brightspace_df

            st.success(f"Brightspace data refreshed successfully. Last refreshed: {st.session_state.last_sync}")

            st.markdown("### Evidence Sources Retrieved")

            check_col1, check_col2 = st.columns(2)

            with check_col1:
                st.markdown("✅ Assignment letter grades")
                st.markdown("✅ Professor assignment feedback")
                st.markdown("✅ Assignment names and weekly modules")

            with check_col2:
                st.markdown("✅ Anonymous student reflections")
                st.markdown("✅ Confidence scores")
                st.markdown("✅ Career relevance scores")

            st.markdown("---")
            st.markdown("### Preview of Retrieved Evidence")
            st.caption("Preview is limited to the first 10 records. Student identifiers are anonymized.")

            preview_cols = [
                "student_id",
                "week",
                "module_topic",
                "assignment_name",
                "assignment_grade",
                "signal_category",
                "confidence_score",
                "career_relevance_score"
            ]

            st.dataframe(df[preview_cols].head(10), use_container_width=True)

            st.markdown("---")
            st.success("Data is ready for AI learning gap analysis on the Faculty Dashboard.")

        else:
            st.warning("Brightspace evidence has not been refreshed yet.")

    with right_col:
        st.subheader("Sync Summary")

        if st.session_state.brightspace_data_loaded:
            df = st.session_state.brightspace_df

            st.metric("Assignment Records", len(df))
            st.metric("Anonymous Students", df["student_id"].nunique())
            st.metric("Modules Matched", df["week"].nunique())
            st.metric("Feedback Comments", df["professor_feedback"].count())
            st.metric("Student Reflections", df["reflection_text"].count())

            st.markdown("---")

            avg_confidence = round(df["confidence_score"].mean(), 2)
            avg_grade = round(df["assignment_grade"].mean(), 1)

            st.metric("Average Confidence", f"{avg_confidence}/5")
            st.metric("Average Assignment Outcome", f"{avg_grade}%")

        else:
            st.metric("Assignment Records", "0")
            st.metric("Anonymous Students", "0")
            st.metric("Modules Matched", "0")
            st.metric("Feedback Comments", "0")
            st.metric("Student Reflections", "0")

        st.markdown("---")

        st.markdown(
            """
            <div class="small-card">
                <h4>Privacy Guardrail</h4>
                <p>
                Retrieved records use anonymous student IDs. The dashboard will summarize patterns
                at the class and learning-outcome level, not as individual student profiles.
                </p>
            </div>
            """,
            unsafe_allow_html=True
        )
